```python
class ProcessRequest(BaseModel):
    source: str  # "upload" or "youtube"
    file_id: str = ""  # for upload
    youtube_url: str = ""  # for youtube
    config: ProcessingConfig = ProcessingConfig()

@app.post("/process")
async def process_video(req: ProcessRequest):
    """
    Chains: upload/youtube → transcribe → energy → rank
    Returns ranked moments ready for dashboard display.
    """
    # Step 1: Get video path (upload already done, or download youtube)
    if req.source == "youtube":
        result = save_youtube(req.youtube_url)
        video_id = result["video_id"]
    else:
        video_id = req.file_id
    
    info = get_video_info(video_id)
    if not info:
        raise HTTPException(status_code=404, detail="Video not found")
    video_path = info["path"]
    
    # Step 2: Transcribe (if not already done)
    transcript_path = f"/tmp/clipwise/{video_id}/transcript.json"
    if not os.path.exists(transcript_path):
        from services.transcription import transcribe_video as do_transcribe
        do_transcribe(video_path, video_id)
    
    # Step 3: Extract energy (if not already done)
    energy_path = f"/tmp/clipwise/{video_id}/energy.json"
    if not os.path.exists(energy_path):
        energy_profile = get_energy_profile(video_path)
        save_energy_data(video_id, energy_profile["energy_data"], energy_profile["segment_scores"], energy_profile["peak_times"])
    
    # Step 4: Rank moments
    config = {
        "min_clip_duration": req.config.min_clip_duration,
        "max_clip_duration": req.config.max_clip_duration,
        "target_clips": req.config.target_clips,
        "format": req.config.format,
    }
    result = rank_moments(video_id, config, use_llm=True)
    moments_path = save_moments(video_id, result)
    
    return {
        "video_id": video_id,
        "status": "complete",
        "moments_path": moments_path,
        "moment_count": len(result.get("moments", [])),
        "moments": result.get("moments", [])
    }
```

const processVideo = async () => {
  setError(null)
  setLogs([])
  setProcessingState('idle')
  setProgress(0)
  setMoments([])

  const inputSource = activeTab === 'youtube' ? youtubeUrl : file?.name || 'arquivo'

  try {
    setProcessingState('uploading')
    setCurrentStep('Enviando arquivo...')
    addLog(`Iniciando processamento: ${inputSource}`)

    let videoId: string

    // Step 1: Upload or YouTube download
    if (activeTab === 'youtube') {
      setCurrentStep('Baixando YouTube...')
      const ytRes = await fetch('/youtube', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ url: youtubeUrl })
      })
      if (!ytRes.ok) throw new Error('YouTube download failed')
      const ytData = await ytRes.json()
      videoId = ytData.video_id
      addLog('YouTube baixado com sucesso', 'success')
    } else {
      setCurrentStep('Enviando arquivo...')
      const formData = new FormData()
      if (file) formData.append('file', file)
      const uploadRes = await fetch('/upload', { method: 'POST', body: formData })
      if (!uploadRes.ok) throw new Error('Upload failed')
      const uploadData = await uploadRes.json()
      videoId = uploadData.video_id
      addLog('Arquivo enviado com sucesso', 'success')
    }
    setVideoId(videoId)

    // Step 2: Process chain (transcribe → energy → rank)
    setProcessingState('transcribing')
    setCurrentStep('Transcrevendo...')
    setProgress(10)
    addLog('Iniciando transcrição...')

    setProcessingState('analyzing')
    setCurrentStep('Processando...')
    setProgress(50)
    addLog('Analisando momentos...')

    const processRes = await fetch('/process', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        source: activeTab === 'youtube' ? 'youtube' : 'upload',
        file_id: videoId,
        youtube_url: activeTab === 'youtube' ? youtubeUrl : '',
        config: config
      })
    })
    if (!processRes.ok) throw new Error('Processing failed')
    const processData = await processRes.json()
    setProgress(90)

    // Step 3: Fetch moments
    setCurrentStep('Carregando momentos...')
    const momentsRes = await fetch(`/moments/${videoId}`)
    if (!momentsRes.ok) throw new Error('Failed to load moments')
    const momentsData = await momentsRes.json()
    setMoments(momentsData.moments || [])
    setProgress(100)

    setProcessingState('complete')
    setCurrentStep('')
    addLog('Processamento completo!', 'success')

  } catch (err) {
    setProcessingState('error')
    setError(err instanceof Error ? err.message : 'Erro desconhecido')
    addLog(`Erro: ${err instanceof Error ? err.message : 'Erro desconhecido'}`, 'error')
  }
}
```

// Format: [start]-[end] per line (Opus-compatible)
const formatTimestamp = (start: number, end: number) => {
  const fmt = (s: number) => {
    const h = Math.floor(s / 3600)
    const m = Math.floor((s % 3600) / 60)
    const sec = Math.floor(s % 60)
    return `${h.toString().padStart(2, '0')}:${m.toString().padStart(2, '0')}:${sec.toString().padStart(2, '0')}`
  }
  return `${fmt(start)}-${fmt(end)}`
}

const copyTimestamps = async () => {
  const selected = moments.filter(m => m.selected)
  if (selected.length === 0) {
    setError('Selecione pelo menos um momento')
    return
  }
  const text = selected.map(m => formatTimestamp(m.start, m.end)).join('\n')
  await navigator.clipboard.writeText(text)
  // Show toast (add toast state if needed)
  addLog(`${selected.length} timestamps copiados!`, 'success')
}

const exportJSON = () => {
  const selected = moments.filter(m => m.selected)
  const blob = new Blob([JSON.stringify({ moments: selected }, null, 2)], { type: 'application/json' })
  const url = URL.createObjectURL(blob)
  const a = document.createElement('a')
  a.href = url
  a.download = 'moments.json'
  a.click()
  URL.revokeObjectURL(url)
  addLog('JSON exportado!', 'success')
}
```
