```python
import httpx
import os
from typing import Optional

class OpusClient:
    """Opus Clip API client with resumable upload support."""
    
    BASE_URL = "https://api.opus.pro/v1"
    
    def __init__(self, api_key: str):
        self.api_key = api_key
        self.headers = {
            "Authorization": f"Bearer {api_key}",
            "X-Request-ID": str(uuid.uuid4())
        }
    
    async def get_upload_link(self, filename: str, content_type: str = "video/mp4") -> dict:
        """Step 1: Get resumable upload URL and uploadId"""
        response = httpx.post(
            f"{self.BASE_URL}/upload-links",
            headers=self.headers,
            json={"filename": filename, "content_type": content_type}
        )
        response.raise_for_status()
        data = response.json()
        return {"upload_id": data["uploadId"], "url": data["url"]}
    
    async def initiate_resumable_upload(self, url: str, filename: str) -> str:
        """Step 2: Start resumable upload - get location header"""
        response = httpx.post(
            url,
            headers={
                "Authorization": f"Bearer {self.api_key}",
                "X-Goog-Resumable": "start"
            },
            content=""
        )
        # Location header contains the upload location
        return response.headers["Location"]
    
    async def upload_video_chunked(self, location: str, filepath: str, 
                                   progress_callback: Optional[callable] = None) -> bool:
        """Step 3: PUT video in chunks to upload location"""
        file_size = os.path.getsize(filepath)
        chunk_size = 5 * 1024 * 1024  # 5MB chunks
        
        with open(filepath, "rb") as f:
            offset = 0
            while True:
                chunk = f.read(chunk_size)
                if not chunk:
                    break
                
                end = offset + len(chunk) - 1
                headers = {
                    "Content-Range": f"bytes {offset}-{end}/{file_size}",
                    "Content-Length": str(len(chunk))
                }
                
                response = httpx.put(location, headers=headers, content=chunk)
                if response.status_code not in (200, 201):
                    # Retry or handle error
                    pass
                
                offset += len(chunk)
                if progress_callback:
                    progress_callback(offset / file_size * 100)
        
        return True
    
    async def create_clip_project(self, upload_id: str, timestamps: list[dict]) -> dict:
        """Step 4: Create clip project with uploadId + timestamps"""
        response = httpx.post(
            f"{self.BASE_URL}/clip-projects",
            headers=self.headers,
            json={
                "uploadId": upload_id,
                "clips": [{"startTime": ts["start"], "duration": ts["duration"]} for ts in timestamps]
            }
        )
        response.raise_for_status()
        return response.json()
    
    async def send_moments_direct(self, video_url: str, timestamps: list[dict]) -> dict:
        """For YouTube videos - send directly without upload"""
        response = httpx.post(
            f"{self.BASE_URL}/clip-projects",
            headers=self.headers,
            json={
                "videoUrl": video_url,
                "clips": [{"startTime": ts["start"], "duration": ts["duration"]} for ts in timestamps]
            }
        )
        response.raise_for_status()
        return response.json()
    
    async def poll_clip_status(self, project_id: str) -> dict:
        """Poll for clip generation status"""
        response = httpx.get(
            f"{self.BASE_URL}/clip-projects/{project_id}",
            headers=self.headers
        )
        response.raise_for_status()
        return response.json()
    
    async def download_clip(self, clip_url: str, output_path: str) -> str:
        """Download a completed clip"""
        response = httpx.get(clip_url, follow_redirects=True)
        response.raise_for_status()
        
        with open(output_path, "wb") as f:
            f.write(response.content)
        
        return output_path
```

Key behaviors:
- Resumable upload with chunked PUT (required for large files)
- Progress callback during upload
- Direct send for YouTube URLs (skip upload step per D-06)
- Status polling for clip generation

```python
from backend.services.opus_client import OpusClient
from fastapi import HTTPException
import os

# Environment variable for Opus API key
OPUS_API_KEY = os.getenv("OPUS_API_KEY", "")

@app.post("/opus/upload-link")
async def get_upload_link(video_id: str, filename: str, content_type: str = "video/mp4"):
    """Get a resumable upload URL for a video file."""
    if not OPUS_API_KEY:
        raise HTTPException(status_code=500, detail="OPUS_API_KEY not configured")
    
    client = OpusClient(OPUS_API_KEY)
    
    # Get upload link
    result = await client.get_upload_link(filename, content_type)
    
    # Store upload_id in temp dict (production: use Redis or DB)
    upload_state[video_id] = {"upload_id": result["upload_id"], "url": result["url"]}
    
    return {"upload_id": result["upload_id"], "url": result["url"]}


@app.post("/opus/upload-complete")
async def complete_upload(video_id: str, upload_id: str):
    """Confirm upload is complete and enable timestamp sending."""
    # Mark upload as complete in state
    upload_state[video_id]["complete"] = True
    return {"status": "complete", "video_id": video_id}


@app.post("/opus/send-moments")
async def send_moments_to_opus(video_id: str, moments: list[dict], use_youtube: bool = False):
    """
    Send moments to Opus Clip API.
    
    If use_youtube=True, video is from YouTube (skip upload, use direct URL).
    Otherwise, use the uploaded video via upload_id.
    """
    if not OPUS_API_KEY:
        raise HTTPException(status_code=500, detail="OPUS_API_KEY not configured")
    
    client = OpusClient(OPUS_API_KEY)
    
    timestamps = [{"start": m["start"], "duration": m["duration"]} for m in moments]
    
    if use_youtube:
        # YouTube video - send directly
        video_url = upload_state.get(video_id, {}).get("youtube_url")
        if not video_url:
            raise HTTPException(status_code=400, detail="No YouTube URL for this video_id")
        
        result = await client.send_moments_direct(video_url, timestamps)
    else:
        # Uploaded video - use upload_id
        upload_data = upload_state.get(video_id, {})
        if not upload_data.get("complete"):
            raise HTTPException(status_code=400, detail="Video upload not complete")
        
        result = await client.create_clip_project(upload_data["upload_id"], timestamps)
    
    # Store job_id for polling
    upload_state[video_id]["job_id"] = result.get("jobId") or result.get("id")
    upload_state[video_id]["moments_sent"] = len(moments)
    
    return {"job_id": result.get("jobId") or result.get("id"), "status": "sent"}


@app.get("/opus/status/{job_id}")
async def get_clip_status(job_id: str):
    """Poll for clip generation status."""
    if not OPUS_API_KEY:
        raise HTTPException(status_code=500, detail="OPUS_API_KEY not configured")
    
    client = OpusClient(OPUS_API_KEY)
    status = await client.poll_clip_status(job_id)
    
    return {
        "job_id": job_id,
        "status": status.get("status"),  # "processing", "ready", "failed"
        "clips": status.get("clips", []),  # [{clipId, url, status}]
    }


# In-memory state (replace with Redis/DB in production)
upload_state: dict = {}
```

```tsx
"use client";

import { useState } from "react";
import { Upload, CheckCircle, AlertCircle } from "lucide-react";

interface OpusUploadButtonProps {
  videoId: string;
  videoFilename?: string;
  isYoutube?: boolean;
  youtubeUrl?: string;
  onUploadComplete?: (uploadId: string) => void;
  disabled?: boolean;
}

export default function OpusUploadButton({
  videoId,
  videoFilename,
  isYoutube = false,
  youtubeUrl,
  onUploadComplete,
  disabled = false,
}: OpusUploadButtonProps) {
  const [status, setStatus] = useState<"idle" | "uploading" | "success" | "error">("idle");
  const [progress, setProgress] = useState(0);
  const [errorMessage, setErrorMessage] = useState("");

  const handleUpload = async () => {
    if (isYoutube && youtubeUrl) {
      // For YouTube: skip upload, directly enable timestamp sending
      setStatus("uploading");
      try {
        // Store YouTube URL for backend reference
        await fetch("/api/opus/store-youtube", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ video_id: videoId, youtube_url: youtubeUrl }),
        });
        setStatus("success");
        onUploadComplete?.("youtube");
      } catch (err) {
        setStatus("error");
        setErrorMessage("Erro ao configurar vídeo do YouTube");
      }
      return;
    }

    if (!videoFilename) return;

    setStatus("uploading");
    setProgress(0);

    try {
      // Step 1: Get upload link
      const linkRes = await fetch("/api/opus/upload-link", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ video_id: videoId, filename: videoFilename }),
      });
      
      if (!linkRes.ok) throw new Error("Falha ao obter link de upload");
      
      const { upload_id, url } = await linkRes.json();

      // Step 2: Initiate resumable upload
      const startRes = await fetch(url, {
        method: "POST",
        headers: {
          "Authorization": `Bearer ${process.env.NEXT_PUBLIC_OPUS_API_KEY}`,
          "X-Goog-Resumable": "start",
        },
      });
      
      const location = startRes.headers.get("Location");
      if (!location) throw new Error("Não foi possível iniciar upload");

      // Step 3: Upload video in chunks with progress
      const videoRes = await fetch(`/api/opus/video/${videoId}`);
      const videoBlob = await videoRes.blob();
      
      // Simple chunked upload with progress simulation
      const chunkSize = 5 * 1024 * 1024;
      const totalChunks = Math.ceil(videoBlob.size / chunkSize);
      let uploaded = 0;

      for (let i = 0; i < totalChunks; i++) {
        const start = i * chunkSize;
        const end = Math.min(start + chunkSize, videoBlob.size);
        const chunk = videoBlob.slice(start, end);
        
        // Upload chunk to location URL
        await fetch(location, {
          method: "PUT",
          headers: {
            "Content-Range": `bytes ${start}-${end - 1}/${videoBlob.size}`,
            "Content-Length": String(end - start),
          },
          body: chunk,
        });
        
        uploaded++;
        setProgress((uploaded / totalChunks) * 100);
      }

      // Step 4: Confirm upload complete
      await fetch("/api/opus/upload-complete", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ video_id: videoId, upload_id }),
      });

      setStatus("success");
      onUploadComplete?.(upload_id);
    } catch (err) {
      setStatus("error");
      setErrorMessage("Erro ao enviar vídeo para o Opus. Verifique sua chave de API.");
    }
  };

  if (status === "success") {
    return (
      <div className="flex items-center gap-2 p-4 rounded-lg bg-green-900/20 border border-green-700">
        <CheckCircle className="w-5 h-5 text-green-500" />
        <span className="text-green-400">Vídeo enviado! Agora você pode enviar os momentos.</span>
      </div>
    );
  }

  return (
    <div className="space-y-2">
      <button
        onClick={handleUpload}
        disabled={disabled || status === "uploading"}
        className={`
          w-full flex items-center justify-center gap-2 px-4 py-3 rounded-lg font-medium
          transition-colors
          ${disabled ? "bg-gray-600 cursor-not-allowed" : "bg-blue-600 hover:bg-blue-500"}
          ${status === "uploading" ? "bg-blue-700" : ""}
        `}
      >
        {status === "uploading" ? (
          <>
            <Upload className="w-4 h-4 animate-spin" />
            Enviando vídeo para o Opus...
          </>
        ) : (
          <>
            <Upload className="w-4 h-4" />
            Subir vídeo para Opus
          </>
        )}
      </button>

      {status === "uploading" && (
        <div className="w-full bg-gray-700 rounded-full h-2">
          <div
            className="bg-blue-500 h-2 rounded-full transition-all"
            style={{ width: `${progress}%` }}
          />
        </div>
      )}

      {status === "error" && (
        <div className="flex items-center gap-2 p-3 rounded-lg bg-red-900/20 border border-red-700">
          <AlertCircle className="w-4 h-4 text-red-500" />
          <span className="text-red-400 text-sm">{errorMessage}</span>
        </div>
      )}

      <p className="text-xs text-gray-400 text-center">
        {isYoutube ? "Vídeo do YouTube - upload será ignorado" : "O envio pode demorar dependendo do tamanho do arquivo"}
      </p>
    </div>
  );
}
```
